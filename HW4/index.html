Chapter 1: Introduction to System Programming

1.1 What is System Programming

System programming involves writing software that directly interacts with the operating system kernel, manages hardware resources, and provides low-level services. Unlike application programming, system programming requires in-depth understanding of the OS internals.

Scope of System Programming

  • OS Kernel: Process scheduling, memory management, file systems
  • Compilers & Assemblers: Translating high-level languages to machine code
  • Linkers & Loaders: Combining object files into executables
  • Device Drivers: Controlling hardware devices
  • Utility Programs: Shells, system call libraries

1.2 System Programming vs Application Programming

CharacteristicSystem ProgrammingApplication Programming
Execution ModeKernel mode / User modeUser mode
Resource AccessDirect hardware accessVia system calls
LanguagesC / Rust / AssemblyAny language
Error HandlingMust handle low-level errorsCan rely on OS abstractions
PerformanceExtremely highDepends on requirements

Chapter 2: Process Management

2.1 Process Concept

A process is an instance of a running program. Each process has its own:

  • Address space: code, data, stack, heap
  • System resources: file descriptors, signal handlers
  • Execution state: register values, program counter

2.2 Process States

Created → Ready ↔ Running → Terminated
            ↓
         Blocked
StateDescription
RunningCurrently executing on the CPU
ReadyWaiting for CPU scheduling
BlockedWaiting for I/O or an event
ZombieTerminated but parent hasn't reaped
OrphanParent terminated first

2.3 fork() — Creating Processes

#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>

int main() {
    pid_t pid = fork();
    
    if (pid < 0) {
        perror("fork failed");
        exit(1);
    }
    
    if (pid == 0) {
        printf("[Child] PID: %d, Parent PID: %d\n", getpid(), getppid());
    } else {
        printf("[Parent] PID: %d, Child PID: %d\n", getpid(), pid);
    }
    
    return 0;
}
  • Parent receives the child's PID
  • Child receives 0
  • Child copies parent's address space (Copy-on-Write)

2.4 exec() — Executing Programs

#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();
    
    if (pid == 0) {
        execlp("ls", "ls", "-la", NULL);
        perror("exec failed");
        return 1;
    }
    
    int status;
    waitpid(pid, &status, 0);
    printf("Child exited, status: %d\n", WEXITSTATUS(status));
    
    return 0;
}
FunctionDescription
execlp()Search PATH, argument list
execvp()Search PATH, argument array
execle()Specify path, pass environment
execvpe()Specify path + environment variables

2.5 wait() and waitpid()

#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();
    
    if (pid == 0) {
        printf("Child working...\n");
        sleep(2);
        printf("Child done\n");
        return 42;
    }
    
    int status;
    wait(&status);
    
    if (WIFEXITED(status)) {
        printf("Child exited normally, return: %d\n", WEXITSTATUS(status));
    }
    
    return 0;
}

2.6 Zombie and Orphan Processes

#include <stdio.h>

#include <stdlib.h>

int main() {

Chapter 3: Memory Management

3.1 Virtual Memory

The OS provides each process with an independent virtual address space. The MMU (Memory Management Unit) maps virtual addresses to physical memory via page tables.

Virtual Address → [MMU + Page Table] → Physical Address

Address Space Layout

  High Addr +--------------------+
            | Environment/Args    |
            |--------------------|
            | Stack (grows down)  |
            |        ↓            |
            |        ↑            |
            | Heap (grows up)     |
            |--------------------|
            | BSS (uninitialized) |
            | Data (initialized)  |
            |--------------------|
            | Text (code)         |
  Low Addr  +--------------------+

3.2 Dynamic Memory Allocation

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    int *arr = (int *)malloc(5 * sizeof(int));
    if (arr == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        return 1;
    }
    
    for (int i = 0; i < 5; i++) arr[i] = i * 10;
    
    int *bigger = (int *)realloc(arr, 10 * sizeof(int));
    if (bigger == NULL) { free(arr); return 1; }
    
    for (int i = 5; i < 10; i++) bigger[i] = i * 10;
    
    int *zero = (int *)calloc(5, sizeof(int));
    
    free(bigger);
    free(zero);
    
    return 0;
}

3.3 sbrk() and brk()

Under the hood, malloc uses sbrk()/brk() to adjust the process's heap boundary:

#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>

int main() {
    void *before = sbrk(0);
    printf("Heap boundary before: %p\n", before);
    
    void *p = malloc(1024 * 1024);
    void *after = sbrk(0);
    printf("Heap boundary after: %p\n", after);
    printf("Difference: %ld bytes\n", (long)(after - before));
    
    free(p);
    return 0;
}

3.4 mmap() — Memory Mapping

#include <stdio.h>
#include <sys/mman.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

int main() {
    size_t size = 4096;
    
    void *addr = mmap(NULL, size,
                      PROT_READ | PROT_WRITE,
                      MAP_PRIVATE | MAP_ANONYMOUS,
                      -1, 0);
    
    if (addr == MAP_FAILED) {
        perror("mmap failed");
        return 1;
    }
    
    strcpy((char *)addr, "Hello, mmap!");
    printf("Content: %s\n", (char *)addr);
    
    munmap(addr, size);
    return 0;
}
FlagDescription
PROT_READReadable
PROT_WRITEWritable
PROT_EXECExecutable
MAP_PRIVATEPrivate mapping (Copy-on-Write)
MAP_SHAREDShared mapping
MAP_ANONYMOUSAnonymous mapping (no backing file)

Chapter 4: Threads & Concurrency

4.1 Introduction to Threads

A thread is the smallest unit of CPU scheduling. Threads within the same process share memory but have their own stack.

Process
├── Address Space (code, data, heap)
├── Thread 1 (stack, registers)
├── Thread 2 (stack, registers)
└── Thread 3 (stack, registers)

4.2 Basic pthread Usage

#include <stdio.h>
#include <pthread.h>
#include <unistd.h>

void *thread_func(void *arg) {
    int id = *(int *)arg;
    printf("Thread %d started, PID: %d\n", id, getpid());
    sleep(1);
    printf("Thread %d done\n", id);
    return NULL;
}

int main() {
    pthread_t t1, t2;
    int id1 = 1, id2 = 2;
    
    pthread_create(&t1, NULL, thread_func, &id1);
    pthread_create(&t2, NULL, thread_func, &id2);
    
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    
    printf("Main thread done\n");
    return 0;
}
gcc -pthread -o thread thread.c

4.3 Race Conditions

#include <stdio.h>
#include <pthread.h>

int counter = 0;

void *increment(void *arg) {
    for (int i = 0; i < 1000000; i++) {
        counter++;  // NOT atomic!
    }
    return NULL;
}

int main() {
    pthread_t t1, t2;
    pthread_create(&t1, NULL, increment, NULL);
    pthread_create(&t2, NULL, increment, NULL);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    
    printf("Expected: 2000000, Actual: %d\n", counter);
    return 0;
}
counter++ is not atomic — it involves read, modify, and write steps.

4.4 Mutex (Mutual Exclusion)

#include <stdio.h>
#include <pthread.h>

int counter = 0;
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

void *increment(void *arg) {
    for (int i = 0; i < 1000000; i++) {
        pthread_mutex_lock(&mutex);
        counter++;
        pthread_mutex_unlock(&mutex);
    }
    return NULL;
}

int main() {
    pthread_t t1, t2;
    pthread_create(&t1, NULL, increment, NULL);
    pthread_create(&t2, NULL, increment, NULL);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    
    printf("Expected: 2000000, Actual: %d\n", counter);
    pthread_mutex_destroy(&mutex);
    return 0;
}

4.5 Deadlock

pthread_mutex_t lock1 = PTHREAD_MUTEX_INITIALIZER;
pthread_mutex_t lock2 = PTHREAD_MUTEX_INITIALIZER;

void *thread_a(void *arg) {
    pthread_mutex_lock(&lock1);
    sleep(1);
    pthread_mutex_lock(&lock2);  // waits forever
    ...
}

void *thread_b(void *arg) {
    pthread_mutex_lock(&lock2);
    sleep(1);
    pthread_mutex_lock(&lock1);  // waits forever
    ...
}

Deadlock Prevention

  1. Fixed lock ordering: all threads acquire locks in the same order
  2. Try-lock: use pthread_mutex_trylock()
  3. Reduce lock scope: protect only the necessary code

4.6 Condition Variables

pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
pthread_cond_t cond = PTHREAD_COND_INITIALIZER;
int ready = 0;

void *worker(void *arg) {
    pthread_mutex_lock(&mutex);
    while (!ready) pthread_cond_wait(&cond, &mutex);
    printf("Worker: received notification, starting work\n");
    pthread_mutex_unlock(&mutex);
    return NULL;
}

void *notifier(void *arg) {
    sleep(1);
    pthread_mutex_lock(&mutex);
    ready = 1;
    pthread_cond_signal(&cond);
    pthread_mutex_unlock(&mutex);
    return NULL;
}

4.7 Thread-Safe Queue Example

typedef struct {
    int *buf; int size, head, tail;
    pthread_mutex_t mutex;
    pthread_cond_t not_full, not_empty;
} queue_t;

queue_t *queue_create(int size) {
    queue_t *q = malloc(sizeof(queue_t));
    q->buf = malloc(size * sizeof(int));
    q->size = size; q->head = q->tail = 0;
    pthread_mutex_init(&q->mutex, NULL);
    pthread_cond_init(&q->not_full, NULL);
    pthread_cond_init(&q->not_empty, NULL);
    return q;
}

void queue_push(queue_t *q, int val) {
    pthread_mutex_lock(&q->mutex);
    while ((q->tail + 1) % q->size == q->head)
        pthread_cond_wait(&q->not_full, &q->mutex);
    q->buf[q->tail] = val;
    q->tail = (q->tail + 1) % q->size;
    pthread_cond_signal(&q->not_empty);
    pthread_mutex_unlock(&q->mutex);
}

int queue_pop(queue_t *q) {
    pthread_mutex_lock(&q->mutex);
    while (q->head == q->tail)
        pthread_cond_wait(&q->not_empty, &q->mutex);
    int val = q->buf[q->head];
    q->head = (q->head + 1) % q->size;
    pthread_cond_signal(&q->not_full);
    pthread_mutex_unlock(&q->mutex);
    return val;
}

Summary

  • Threads share memory but have independent stacks
  • Mutexes protect shared resources against race conditions
  • Incorrect lock ordering causes deadlocks
  • Condition variables enable thread notification

Chapter 5: File I/O

5.1 File Descriptors

In Linux, all I/O operations use file descriptors — non-negative integers the kernel uses to track open files.

stdin  (0) ← keyboard
stdout (1) → screen
stderr (2) → screen

5.2 open() / close()

#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdlib.h>

int main() {
    int fd = open("test.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    
    if (fd < 0) {
        perror("open failed");
        exit(1);
    }
    
    printf("File descriptor: %d\n", fd);
    close(fd);
    return 0;
}
FlagDescription

O_RDWRRead and write

5.3 read() / write()

#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <string.h>
#include <stdlib.h>

int main() {
    const char *msg = "Hello, System Programming!\n";
    
    int fd = open("output.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd < 0) { perror("open failed"); return 1; }
    
    ssize_t written = write(fd, msg, strlen(msg));
    printf("Wrote %zd bytes\n", written);
    close(fd);
    
    fd = open("output.txt", O_RDONLY);
    char buf[256] = {0};
    
    ssize_t bytes = read(fd, buf, sizeof(buf) - 1);
    printf("Read %zd bytes: %s", bytes, buf);
    close(fd);
    
    return 0;
}

5.4 dup() / dup2()

dup2() duplicates file descriptors, commonly used for redirection:

#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdlib.h>

int main() {
    int fd = open("redirect.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd < 0) { perror("open failed"); return 1; }
    
    dup2(fd, STDOUT_FILENO);
    close(fd);
    
    printf("This goes to redirect.txt\n");
    printf("stdout has been redirected\n");
    
    return 0;
}

5.5 Pipe

#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
#include <string.h>
#include <stdlib.h>

int main() {
    int pipefd[2];
    if (pipe(pipefd) < 0) { perror("pipe failed"); return 1; }
    
    pid_t pid = fork();
    
    if (pid == 0) {
        close(pipefd[0]);
        const char *msg = "Hello from child!";
        write(pipefd[1], msg, strlen(msg));
        close(pipefd[1]);
        exit(0);
    }
    
    close(pipefd[1]);
    char buf[256] = {0};
    read(pipefd[0], buf, sizeof(buf));
    printf("Parent received: %s\n", buf);
    close(pipefd[0]);
    
    wait(NULL);
    return 0;
}

5.6 Shell Pipeline (simulating ls | wc -l)

int main() {
    int pipefd[2];
    pipe(pipefd);
    
    pid_t pid1 = fork();
    if (pid1 == 0) {
        close(pipefd[0]);
        dup2(pipefd[1], STDOUT_FILENO);
        close(pipefd[1]);
        execlp("ls", "ls", NULL);
        perror("exec ls failed");
        exit(1);
    }
    
    pid_t pid2 = fork();
    if (pid2 == 0) {
        close(pipefd[1]);
        dup2(pipefd[0], STDIN_FILENO);
        close(pipefd[0]);
        execlp("wc", "wc", "-l", NULL);
        perror("exec wc failed");
        exit(1);
    }
    
    close(pipefd[0]); close(pipefd[1]);
    waitpid(pid1, NULL, 0); waitpid(pid2, NULL, 0);
    return 0;
}

5.7 lseek() — File Positioning

int fd = open("seek.txt", O_RDWR | O_CREAT | O_TRUNC, 0644);

lseek(fd, 0, SEEK_SET);  // move to start

6.2 Bidirectional Pipes

int to_child[2], to_parent[2];
pipe(to_child); pipe(to_parent);

pid_t pid = fork();

if (pid == 0) {
    close(to_child[1]); close(to_parent[0]);
    char msg[256];
    read(to_child[0], msg, sizeof(msg));
    printf("[Child] Received: %s\n", msg);
    write(to_parent[1], "OK", 2);
    close(to_child[0]); close(to_parent[1]);
    exit(0);
}

close(to_child[0]); close(to_parent[1]);
write(to_child[1], "Hello", 5);

char reply[256];
read(to_parent[0], reply, sizeof(reply));
printf("[Parent] Reply: %s\n", reply);

wait(NULL);

6.3 FIFO (Named Pipe)

// writer.c
const char *fifo_path = "/tmp/my_fifo";
mkfifo(fifo_path, 0666);

printf("Waiting for reader...\n");
int fd = open(fifo_path, O_WRONLY);

printf("Received: %s\n", buf);

./reader &   # start reader (background)
./writer     # start writer

6.6 Message Queues

#include <sys/msg.h>
#include <sys/ipc.h>

struct msg_buf { long mtype; char mtext[256]; };

key_t key = IPC_PRIVATE;
int msqid = msgget(key, IPC_CREAT | 0666);

pid_t pid = fork();

if (pid == 0) {

6.7 Semaphores

#include <sys/sem.h>
#include <sys/ipc.h>

void sem_op(int semid, int op) {
    struct sembuf sb = {0, op, 0};
    semop(semid, &sb, 1);
}

key_t key = IPC_PRIVATE;
int semid = semget(key, 1, IPC_CREAT | 0666);

union semun { int val; struct semid_ds *buf; unsigned short *array; } su;
su.val = 1;
semctl(semid, 0, SETVAL, su);

pid_t pid = fork();
for (int i = 0; i < 5; i++) {
    sem_op(semid, -1);  // P operation (wait)
    printf("Process %d entering critical section\n", getpid());
    sleep(1);
    printf("Process %d leaving critical section\n", getpid());
    sem_op(semid, 1);   // V operation (release)
}

wait(NULL);
semctl(semid, 0, IPC_RMID);

Summary

  • Pipes are suitable for simple parent-child communication
  • FIFO works across unrelated processes
  • Shared memory is the fastest IPC method
  • Message queues provide structured communication
  • Semaphores are used for synchronization and mutual exclusion

7.2 TCP Server

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>

int main() {
    int server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd < 0) { perror("socket failed"); return 1; }
    
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
    
    struct sockaddr_in addr = {
        .sin_family = AF_INET,
        .sin_port = htons(8080),
        .sin_addr.s_addr = INADDR_ANY
    };
    
    if (bind(server_fd, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
        perror("bind failed"); return 1;
    }
    
    if (listen(server_fd, 5) < 0) { perror("listen failed"); return 1; }
    
    printf("Server listening on port 8080...\n");
    
    struct sockaddr_in client_addr;
    socklen_t client_len = sizeof(client_addr);
    
    int client_fd = accept(server_fd, (struct sockaddr *)&client_addr, &client_len);
    if (client_fd < 0) { perror("accept failed"); return 1; }
    
    char buf[1024] = {0};
    read(client_fd, buf, sizeof(buf));
    printf("Received: %s\n", buf);
    
    const char *resp = "Hello from server!";
    write(client_fd, resp, strlen(resp));
    
    close(client_fd);
    close(server_fd);
    return 0;
}

7.3 TCP Client

#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

int main() {
    int sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock < 0) { perror("socket failed"); return 1; }
    
    struct sockaddr_in addr = {
        .sin_family = AF_INET,
        .sin_port = htons(8080)
    };
    inet_pton(AF_INET, "127.0.0.1", &addr.sin_addr);
    
    if (connect(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
        perror("connect failed"); return 1;
    }
    
    const char *msg = "Hello from client!";
    write(sock, msg, strlen(msg));
    
    char buf[1024] = {0};
    read(sock, buf, sizeof(buf));
    printf("Server response: %s\n", buf);
    
    close(sock);
    return 0;
}

7.4 Concurrent Server (Multi-process)

#include <signal.h>

void handle_client(int client_fd) {
    char buf[1024];
    int n = read(client_fd, buf, sizeof(buf) - 1);
    buf[n] = '\0';
    printf("Process %d handling: %s\n", getpid(), buf);
    
    const char *resp = "HTTP/1.1 200 OK\r\nContent-Length: 13\r\n\r\nHello, World!";
    write(client_fd, resp, strlen(resp));
    close(client_fd);
    exit(0);
}

int main() {
    signal(SIGCHLD, SIG_IGN);
    
    int server_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
    
    struct sockaddr_in addr = {
        .sin_family = AF_INET,
        .sin_port = htons(8080),
        .sin_addr.s_addr = INADDR_ANY
    };
    
    bind(server_fd, (struct sockaddr *)&addr, sizeof(addr));
    listen(server_fd, 10);
    
    printf("Concurrent server started (PID: %d)\n", getpid());
    
    while (1) {
        struct sockaddr_in client;
        socklen_t len = sizeof(client);
        int client_fd = accept(server_fd, (struct sockaddr *)&client, &len);
        
        if (fork() == 0) {
            close(server_fd);
            handle_client(client_fd);
        }
        close(client_fd);
    }
    
    close(server_fd);
    return 0;
}

7.6 select() — Multiplexed I/O

#include <sys/select.h>

int server_fd = socket(AF_INET, SOCK_STREAM, 0);
// ... bind, listen ...

fd_set read_fds;
int max_fd = server_fd;

while (1) {
    FD_ZERO(&read_fds);
    FD_SET(server_fd, &read_fds);
    
    struct timeval tv = {5, 0};
    int ret = select(max_fd + 1, &read_fds, NULL, NULL, &tv);
    
    if (ret < 0) { perror("select error"); break; }
    else if (ret == 0) { printf("Waiting for connection...\n"); continue; }
    
    if (FD_ISSET(server_fd, &read_fds)) {
        struct sockaddr_in client;
        socklen_t len = sizeof(client);
        int client_fd = accept(server_fd, (struct sockaddr *)&client, &len);
        
        char buf[1024];
        int n = read(client_fd, buf, sizeof(buf) - 1);
        buf[n] = '\0';
        printf("Received: %s", buf);
        write(client_fd, "OK\n", 3);
        close(client_fd);
    }
}

7.7 HTTP Request Parsing Example

void handle_http(int client_fd) {
    char buf[4096];
    int n = read(client_fd, buf, sizeof(buf) - 1);
    buf[n] = '\0';
    
    char method[16], path[256];
    sscanf(buf, "%s %s", method, path);
    printf("Method: %s, Path: %s\n", method, path);
    
    char *body = "<html><body><h1>Hello!</h1></body></html>";
    char resp[1024];
    snprintf(resp, sizeof(resp),
             "HTTP/1.1 200 OK\r\n"
             "Content-Type: text/html\r\n"
             "Content-Length: %zu\r\n"
             "\r\n"
             "%s",
             strlen(body), body);
    
    write(client_fd, resp, strlen(resp));
    close(client_fd);
}

  • UDP is lightweight and connectionless, suitable for real-time communication

Chapter 8: Assembly Language, Linking & Loading

8.1 Why Learn Assembly

  • Debugging: reading compiler-generated assembly aids debugging
  • Performance optimization: identify critical paths the compiler missed
  • Low-level control: some OS operations (context switching) require assembly
  • Security analysis: understanding buffer overflow vulnerabilities

8.2 x86-64 Basics (AT&T Syntax)

# Data movement
mov %rax, %rbx       # rax → rbx
mov $42, %rax        # immediate 42 → rax
mov (%rbx), %rax     # memory at rbx → rax

# Arithmetic
add %rcx, %rax       # rax = rax + rcx
sub $1, %rax         # rax = rax - 1
imul %rbx, %rax      # rax = rax * rbx

# Comparison and jumps
cmp $0, %rax         # compare rax with 0
je label             # jump if equal
jne label            # jump if not equal
jg label             # jump if greater
jl label             # jump if less

# Stack operations
push %rbp            # push rbp onto stack
pop %rbp             # pop from stack to rbp
call function        # call function
ret                  # return from function

8.3 Inline Assembly in C

#include <stdio.h>

int main() {
    int a = 10, b = 20, result;
    
    asm volatile (
        "mov %1, %%eax\n"
        "add %2, %%eax\n"
        "mov %%eax, %0\n"
        : "=r" (result)      // output operand
        : "r" (a), "r" (b)   // input operands
        : "%eax"             // clobbered register
    );
    
    printf("%d + %d = %d\n", a, b, result);
    return 0;
}

8.5 Compilation Process

Source (.c) → [Preprocessor] → Expanded source (.i)
            → [Compiler]     → Assembly (.s)
            → [Assembler]    → Object file (.o)
            → [Linker]       → Executable
gcc -E program.c -o program.i     # Preprocess
gcc -S program.i -o program.s     # Compile to assembly
gcc -c program.s -o program.o     # Assemble to object file
gcc program.o -o program          # Link to executable

# Or all at once
gcc program.c -o program

8.6 ELF Format

Linux uses the ELF (Executable and Linkable Format):

file program
readelf -h program     # ELF header
readelf -S program     # Section table
readelf -s program     # Symbol table
objdump -d program     # Disassembly
ELF Header
├── Program Header Table (for execution)
└── Section Header Table (for linking)
    ├── .text    — Code
    ├── .data    — Initialized data
    ├── .bss     — Uninitialized data
    ├── .rodata  — Read-only data
    ├── .symtab  — Symbol table
    └── .strtab  — String table
